% Here is a sample format for dissertation in economics.  You should
% the University of Pennsylvania Doctoral Dissertation Manual to adjust
% any changes they made at the webpage:

% http://guides.library.upenn.edu/dissertation_manual/formatting

\documentclass[12pt]{report}

\input{preamble}

\begin{document}

    % title
    \input{title.tex}
    \newpage

    % copyright
    \input{copyright.tex}
    \newpage

    % dedication
    \input{dedication.tex}
    \newpage

    % acknowledgment
    \input{acknowledgement.tex}
    \newpage

    % abstract
    \input{abstract.tex}
    \newpage

    % table of contents
    \singlespaced
    \tableofcontents
    \newpage

    % list of tables
    \listoftables
    \newpage

    % list of figures
    \listoffigures
    \newpage

    \draftspaced
    \pagenumbering{arabic}

    \chapter*{Introduction}
    \addcontentsline{toc}{chapter}{Introduction}
    \input{introduction.tex}
    \newpage

    \chapter{Title of the chapter}
    \input{chapter1.tex}
    \newpage

    \chapter{Title of the chapter}
    \input{chapter2.tex}
    \newpage

    % appendices
    \input{appendix.tex}
    \newpage

    % bibliography
    \bibliographystyle{plainnat}
    \bibliography{BibFile.bib}

\end{document}
